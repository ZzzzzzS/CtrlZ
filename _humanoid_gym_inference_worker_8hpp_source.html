140 size_t t = this->Scheduler->getTimeStamp();
141 InferencePrecision phase = this->dt * static_cast<InferencePrecision>(t) / this->cycle_time;
142 InferencePrecision clock_sin = std::sin(phase * 2 * M_PI);
143 InferencePrecision clock_cos = std::cos(phase * 2 * M_PI);
144 z::math::Vector<InferencePrecision, 2> ClockVector = { clock_sin, clock_cos };
145 this->Scheduler->template SetData<concat(NetName, "NetClockVector")>(ClockVector);
146
147 auto SingleInputVecScaled = math::cat(
148 ClockVector,
149 UserCmd3,
150 CurrentMotorPos,
151 CurrentMotorVel,
152 LastAction,
153 AngVel,
154 Ang
155 ) * this->InputScaleVec;
156
157 this->HistoryInputBuffer.push(SingleInputVecScaled);
158
159
160 math::Vector<InferencePrecision, INPUT_TENSOR_LENGTH> InputVec;
161 for (size_t i = 0; i < INPUT_STUCK_LENGTH; i++)
162 {
163 std::copy(this->HistoryInputBuffer[i].begin(), this->HistoryInputBuffer[i].end(), InputVec.begin() + i * INPUT_TENSOR_LENGTH_UNIT);
164 }

166 this->InputTensor.Array() = decltype(InputVec)::clamp(InputVec, -this->ClipObservation, this->ClipObservation);

173 void PostProcess() override
174 {
175 auto LastAction = this->OutputTensor.toVector();
176 auto ClipedLastAction = MotorValVec::clamp(LastAction, -this->ClipAction, this->ClipAction);
177 this->Scheduler->template SetData<concat(NetName, "NetLastAction")>(ClipedLastAction);
178
179 auto ScaledAction = ClipedLastAction * this->OutputScaleVec + this->JointDefaultPos;
180 this->Scheduler->template SetData<concat(NetName, "NetScaledAction")>(ScaledAction);
181
182 auto clipedAction = MotorValVec::clamp(ScaledAction, this->JointClipLower, this->JointClipUpper);
183 this->Scheduler->template SetData<concat(NetName, "Action")>(clipedAction);
184
185 this->end_time = std::chrono::steady_clock::now();
186 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(this->end_time - this->start_time);
187 InferencePrecision inference_time = static_cast<InferencePrecision>(duration.count());
188 this->Scheduler->template SetData<concat(NetName, "InferenceTime")>(inference_time);
189 }

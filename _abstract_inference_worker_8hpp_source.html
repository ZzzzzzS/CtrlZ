171 void TaskCreate() override
172 {
173 //bind input and output tensor
174 if (this->InputOrtTensors__.empty() || this->OutputOrtTensors__.empty())
175 throw(std::runtime_error("InputOrtTensors or OutputOrtTensors is empty, call WarpOrtTensor to create Ort tensors before launch scheduler!"));
176
177 if (this->InputNodeNames__.size() != this->InputOrtTensors__.size())
178 throw(std::runtime_error("InputNodeNames size is not equal to InputOrtTensors size!"));
179 if (this->OutputNodeNames__.size() != this->OutputOrtTensors__.size())
180 throw(std::runtime_error("OutputNodeNames size is not equal to OutputOrtTensors size!"));
181
182 this->IoBinding__ = Ort::IoBinding(this->Session__);
183
184 for (size_t i = 0; i < this->InputNodeNames__.size(); i++)
185 {
186 this->IoBinding__.BindInput(this->InputNodeNames__[i].c_str(), this->InputOrtTensors__[i]);
187 }
188
189 for (size_t i = 0; i < this->OutputNodeNames__.size(); i++)
190 {
191 this->IoBinding__.BindOutput(this->OutputNodeNames__[i].c_str(), this->OutputOrtTensors__[i]);
192 }
193
194 //warm up model
195 if (this->WarmUpModel__)
196 {
197 for (size_t i = 0; i < this->WarmUpCnt__; i++)
198 {
199 InferenceOnce();
200 }
201 }

443 static constexpr size_t calculate_index(Indices... indices) {
444 static_assert(sizeof...(Indices) == Shape::num_dims, "Number of indices must match number of dimensions");
445 std::array<int64_t, Shape::num_dims> indices_array = { indices... };
446
447 for (size_t i = 0;i < Shape::num_dims;i++)
448 {
449 if (indices_array[i] < 0)
450 indices_array[i] += Shape::dims_array[i];
451 }
452
453 // calculate the index
454 size_t index = 0;
455 size_t factor = 1;
456
457 for (int i = Shape::num_dims - 1; i >= 0; i--) {
458 //std::cout << "i: " << i << std::endl;
459 index += indices_array[i] * factor;
460 factor *= Shape::dims_array[i];
461
462 if (indices_array[i] >= Shape::dims_array[i] || indices_array[i] < 0)
463 {
464 throw std::out_of_range("Index out of range");
465 }
466 }
467 //std::cout << "index: " << index << std::endl;
468 return index;

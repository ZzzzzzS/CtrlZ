82 void PreProcess() override
83 {
84 this->start_time = std::chrono::steady_clock::now();
85
86 MotorValVec CurrentMotorVel;
87 this->Scheduler->template GetData<"CurrentMotorVelocity">(CurrentMotorVel);
88
89 MotorValVec CurrentMotorPos;
90 this->Scheduler->template GetData<"CurrentMotorPosition">(CurrentMotorPos);
91 CurrentMotorPos -= this->JointDefaultPos;
92
93 MotorValVec LastAction;
94 this->Scheduler->template GetData<"NetLastAction">(LastAction);
95
96 ValVec3 UserCmd3;
97 this->Scheduler->template GetData<"NetUserCommand3">(UserCmd3);
98 UserCmd3 = UserCmd3 * this->Scales_command3;
99
100 ValVec3 LinVel;
101 this->Scheduler->template GetData<"LinearVelocityValue">(LinVel);
102
103 ValVec3 AngVel;
104 this->Scheduler->template GetData<"AngleVelocityValue">(AngVel);
105
106 ValVec3 Ang;
107 this->Scheduler->template GetData<"AngleValue">(Ang);
108
109 ValVec3 ProjectedGravity = ComputeProjectedGravity(Ang, this->GravityVector);
110 this->Scheduler->template SetData<"NetProjectedGravity">(ProjectedGravity);
111
112 size_t t = this->Scheduler->getTimeStamp();
113 InferencePrecision clock = 2 * M_PI / this->CyctleTime * this->dt * static_cast<InferencePrecision>(t);
114 InferencePrecision clock_sin = std::sin(clock);
115 InferencePrecision clock_cos = std::cos(clock);
116 z::math::Vector<InferencePrecision, 2> ClockVector = { clock_sin, clock_cos };
117 this->Scheduler->template SetData<"NetClockVector">(ClockVector);
118
119 z::math::Vector< InferencePrecision, EXTRA_INPUT_LENGTH> ExtraInputVec = math::cat(
120 UserCmd3, ClockVector
121 );
122
123 auto SingleInputVecScaled = math::cat(
124 AngVel,
125 ProjectedGravity,
126 CurrentMotorPos,
127 CurrentMotorVel,
128 LastAction
129 ) * this->InputScaleVec;
130
131 this->HistoryInputBuffer.push(SingleInputVecScaled);
132
133
134 math::Vector<InferencePrecision, INPUT_TENSOR_LENGTH> InputVec;
135 for (size_t i = 0; i < INPUT_STUCK_LENGTH; i++)
136 {
137 std::copy(this->HistoryInputBuffer[i].begin(), this->HistoryInputBuffer[i].end(), InputVec.begin() + i * INPUT_TENSOR_LENGTH_UNIT);
138 }
139 constexpr size_t EXTRA_OFFSET = INPUT_STUCK_LENGTH * INPUT_TENSOR_LENGTH_UNIT;
140 std::copy(ExtraInputVec.begin(), ExtraInputVec.end(), InputVec.begin() + EXTRA_OFFSET);
141
142 this->InputTensor.Array() = decltype(InputVec)::clamp(InputVec, -this->ClipObservation, this->ClipObservation);
143 }
144
145 void PostProcess() override
146 {
147 auto LastAction = this->OutputTensor.toVector();
148 auto ClipedLastAction = MotorValVec::clamp(LastAction, -this->ClipAction, this->ClipAction);
149 this->Scheduler->template SetData<"NetLastAction">(ClipedLastAction);
150
151 auto ScaledAction = ClipedLastAction * this->OutputScaleVec + this->JointDefaultPos;
152 this->Scheduler->template SetData<"NetScaledAction">(ScaledAction);
153
154 if (BlockOutput && static_cast<int>(this->Scheduler->getTimeStamp()) - static_cast<int>(BlockOutputTargetTimestamp) > 0)
155 {
156 BlockOutput = false;
157 }
158
159 if (!this->BlockOutput)
160 {
161 auto clipedAction = MotorValVec::clamp(ScaledAction, this->JointClipLower, this->JointClipUpper);
162 this->Scheduler->template SetData<"TargetMotorPosition">(clipedAction);
163 }
164
165 this->end_time = std::chrono::steady_clock::now();
166 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(this->end_time - this->start_time);
167 InferencePrecision inference_time = static_cast<InferencePrecision>(duration.count()) / 1000.0;
168 this->Scheduler->template SetData<"InferenceTime">(inference_time);
169 }

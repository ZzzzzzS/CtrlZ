102 size_t t = this->Scheduler->getTimeStamp();

105 InferencePrecision clock_cos = std::cos(clock);

137 auto LastAction = this->OutputTensor.toVector();
138 auto ClipedLastAction = MotorValVec::clamp(LastAction, -this->ClipAction, this->ClipAction);
139 this->Scheduler->template SetData<concat(NetName, "NetLastAction")>(ClipedLastAction);
140
141 auto ScaledAction = ClipedLastAction * this->OutputScaleVec + this->JointDefaultPos;
142 this->Scheduler->template SetData<concat(NetName, "NetScaledAction")>(ScaledAction);
143
144 auto clipedAction = MotorValVec::clamp(ScaledAction, this->JointClipLower, this->JointClipUpper);
145 this->Scheduler->template SetData<concat(NetName, "Action")>(clipedAction);
146
147 this->end_time = std::chrono::steady_clock::now();
148 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(this->end_time - this->start_time);
149 InferencePrecision inference_time = static_cast<InferencePrecision>(duration.count()) / 1000.0;
150 this->Scheduler->template SetData<concat(NetName, "InferenceTime")>(inference_time);

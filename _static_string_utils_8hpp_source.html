47 template <std::size_t N, std::size_t M>
48 constexpr auto concat(const char(&a)[N], const char(&b)[M]) {
49 char result[N + M - 1] = {};
50 for (std::size_t i = 0; i < N - 1; ++i) {
51 result[i] = a[i];
52 }
53 for (std::size_t i = 0; i < M; ++i) {
54 result[N - 1 + i] = b[i];
55 }
56 return result;

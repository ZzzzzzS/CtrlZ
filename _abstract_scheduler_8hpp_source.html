184 void SpinOnce()
185 {
186 if (!this->CanSpin)
187 {
188 std::cout << "Scheduler is not started, start scheduler before call spin function!" << std::endl;
189 return;
190 }
191
192 if (this->CheckFrequency__)
193 this->CheckRuntimeFrequency();
194
195 this->SyncMutex.lock();
196
197 this->TimeStamp++;
198
199 for (auto [taskname, task] : TaskList)
200 {
201 task->cnt++;
202 task->cnt = task->cnt % task->div;
203 if (task->cnt % task->div == 0)
204 {
205 if (task->NewRun && !task->Pause)
206 {
207 std::cout << "Task " << taskname << " is running too slow!" << std::endl;
208 }
209 task->NewRun = true;
210 }
211
212 }
213 this->SyncMutex.unlock();
214 this->SyncLock.notify_all();
215
216 this->run_once(MainThreadTaskBlock);
217 }

223 void Spin()
224 {
225 std::chrono::microseconds sleep_all_time = std::chrono::microseconds(static_cast<long long>(this->spin_dt * 1e6));
226 while (this->CanSpin)
227 {
228 auto start_time = std::chrono::high_resolution_clock::now();
229 this->SpinOnce();
230 auto end_time = std::chrono::high_resolution_clock::now();
231 auto time_cost = std::chrono::duration_cast<std::chrono::microseconds>(end_time - start_time);
232 auto sleep_time = sleep_all_time - time_cost;
233 if (sleep_time > std::chrono::microseconds(0))
234 {
235 std::this_thread::sleep_for(sleep_time);
236 }
237 else
238 {
239 std::cout << "SpinOnce is too slow!" << std::endl;
240 }
241 }

624 void run(const std::string& TaskName)
625 {
626 TCB* tcb = TaskName == MainThreadTaskName ? MainThreadTaskBlock : TaskList[TaskName];
627 tcb->cnt = 0;
628 for (auto worker : tcb->workers)
629 {
630 worker->TaskCreate();
631 fflush(stdout);
632 }
633 tcb->NewRun = true;
634 tcb->isRunning = true;
635 while (tcb->isRunning)
636 {
637 {
638 std::unique_lock<std::mutex> lck(tcb->PauseMutex);
639 tcb->PauseLock.wait(lck, [tcb] {return !tcb->Pause || !tcb->isRunning;});
640 }
641
642 {
643 std::unique_lock<std::mutex> lck(this->SyncMutex);
644 this->SyncLock.wait(lck, [tcb] {return tcb->NewRun || !tcb->isRunning;});
645 }
646
647 if (!tcb->isRunning)
648 break;
649
650 run_once(tcb);
651 }
652 for (auto worker : tcb->workers)
653 {
654 worker->TaskDestroy();
655 }

659 void CheckRuntimeFrequency()
660 {
661 auto t = std::chrono::steady_clock::now();
662 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(t - this->last_time).count();
663 this->last_time = t;
664 float dt = static_cast<float>(duration) / 1000000.0;
665 if (this->HistorySpinDt == 0)
666 this->HistorySpinDt = this->spin_dt; //first time, set to spin_dt
667 else [[likely]]
668 this->HistorySpinDt = this->HistorySpinDt * 0.999 + dt * 0.001;
669
670 if (this->HistorySpinDt > this->spin_dt * 1.4 || this->HistorySpinDt < this->spin_dt * 0.6)
671 {
672 //set to red color
673 std::cout << "\033[31m" << "Scheduler spin dt is not consistent with the set dt, please check your code!" << "\033[0m" << std::endl;
674 std::cout << "Current dt: " << this->HistorySpinDt << ", set dt: " << this->spin_dt << std::endl;
675 }

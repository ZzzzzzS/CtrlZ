201 size_t CreateTaskList(const std::string& TaskName, size_t div, bool MainThreadTask = false)
202 {
203 if (TaskList.find(TaskName) != TaskList.end())
204 {
205 std::cout << "Task " << TaskName << " already exists!" << std::endl;

511 void run(const std::string& TaskName)
512 {
513 TCB* tcb = TaskName == MainThreadTaskName ? MainThreadTaskBlock : TaskList[TaskName];
514 tcb->cnt = 0;
515 for (auto worker : tcb->workers)
516 {
517 worker->TaskCreate();
518 fflush(stdout);
519 }
520 tcb->NewRun = true;
521 tcb->isRunning = true;
522 while (tcb->isRunning)
523 {
524 {
525 std::unique_lock<std::mutex> lck(tcb->PauseMutex);
526 tcb->PauseLock.wait(lck, [tcb] {return !tcb->Pause || !tcb->isRunning;});
527 }
528
529 {
530 std::unique_lock<std::mutex> lck(this->SyncMutex);
531 this->SyncLock.wait(lck, [tcb] {return tcb->NewRun || !tcb->isRunning;});
532 }
533
534 if (!tcb->isRunning)
535 break;
536
537 run_once(tcb);
538 }
539 for (auto worker : tcb->workers)
540 {
541 worker->TaskDestroy();
542 }

546 void CheckRuntimeFrequency()
547 {
548 auto t = std::chrono::steady_clock::now();
549 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(t - this->last_time).count();
550 this->last_time = t;
551 float dt = static_cast<float>(duration) / 1000000.0;
552 if (this->HistorySpinDt == 0)
553 this->HistorySpinDt = this->spin_dt; //first time, set to spin_dt
554 else [[likely]]
555 this->HistorySpinDt = this->HistorySpinDt * 0.999 + dt * 0.001;
556
557 if (this->HistorySpinDt > this->spin_dt * 1.4 || this->HistorySpinDt < this->spin_dt * 0.6)
558 {
559 //set to red color
560 std::cout << "\033[31m" << "Scheduler spin dt is not consistent with the set dt, please check your code!" << "\033[0m" << std::endl;
561 std::cout << "Current dt: " << this->HistorySpinDt << ", set dt: " << this->spin_dt << std::endl;
562 }

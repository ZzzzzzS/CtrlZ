170 size_t CreateTaskList(const std::string& TaskName, size_t div, bool MainThreadTask = false)
171 {

474 void run(const std::string& TaskName)
475 {
476 TCB* tcb = TaskName == MainThreadTaskName ? MainThreadTaskBlock : TaskList[TaskName];
477 tcb->cnt = 0;
478 for (auto worker : tcb->workers)
479 {
480 worker->TaskCreate();
481 }
482 tcb->NewRun = true;
483 tcb->isRunning = true;
484 while (tcb->isRunning)
485 {
486 {
487 std::unique_lock<std::mutex> lck(tcb->PauseMutex);
488 tcb->PauseLock.wait(lck, [tcb] {return !tcb->Pause || !tcb->isRunning;});
489 }
490
491 {
492 std::unique_lock<std::mutex> lck(this->SyncMutex);
493 this->SyncLock.wait(lck, [tcb] {return tcb->NewRun || !tcb->isRunning;});
494 }
495
496 if (!tcb->isRunning)
497 break;
498
499 run_once(tcb);
500 }
501 for (auto worker : tcb->workers)
502 {
503 worker->TaskDestroy();
504 }
505 }
